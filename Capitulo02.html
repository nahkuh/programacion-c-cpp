6. Instrucción condicional if

if (condición) {
    /* cuerpo: se ejecuta si condición ≠ 0 */
}

En C: Falso = 0, Verdadero = cualquier valor ≠ 0.

7. Ejemplos representativos

Ejemplo 1: Salida de texto

#include 
int main(void) {
    printf("Bienvenido a C!\n");
    return 0;
}

Ejemplo 2: Suma de dos enteros

#include 
int main(void) {
    int entero1, entero2, suma;
    printf("Introduzca el primer entero\n");
    scanf("%d", &entero1);
    printf("Introduzca el segundo entero\n");
    scanf("%d", &entero2);
    suma = entero1 + entero2;
    printf("La suma es %d\n", suma);
    return 0;
}

Ejemplo 3: Comparación de dos enteros

#include 
int main(void) {
    int num1, num2;
    scanf("%d%d", &num1, &num2);
    if (num1 == num2) printf("%d es igual que %d\n", num1, num2);
    if (num1 != num2) printf("%d no es igual que %d\n", num1, num2);
    if (num1 < num2)  printf("%d es menor que %d\n", num1, num2);
    if (num1 > num2)  printf("%d es mayor que %d\n", num1, num2);
    if (num1 <= num2) printf("%d es menor o igual que %d\n", num1, num2);
    if (num1 >= num2) printf("%d es mayor o igual que %d\n", num1, num2);
    return 0;
}
